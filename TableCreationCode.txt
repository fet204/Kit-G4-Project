
CREATE TABLE Members (
	FullName VARCHAR,
	Email VARCHAR,
	Phone VARCHAR, 
	DateAdded DATE);

CREATE TABLE Events (
	EventName VARCHAR,
	Location VARCHAR,
	Date DATE,
	Description VARCHAR);

CREATE TABLE Comments ( 
	CommentDate DATE,
	Name VARCHAR,
	WebsiteLink VARCHAR,
	Description VARCHAR);

CREATE TABLE Resources (
	Title VARCHAR,
	WebsiteLink VARCHAR,
	Description VARCHAR,
	DateAdded DATE,
	MemberName VARCHAR);

CREATE TABLE Sales (
	Name VARCHAR,
	Email VARCHAR,
	Item VARCHAR,
	Quantity INTEGER,
	Item2 VARCHAR,
	Quantit INTEGER);

CREATE TABLE Merchandise (
	ItemName VARCHAR,
	Description VARCHAR,
	Units VARCHAR,
	PricePerUnit NUMERIC(12,2),
	Quantity INTEGER,
	EntryDate DATE);

CREATE TABLE Donations (
	Name VARCHAR,
	Email VARCHAR,
	Date DATE,
	PriceAmount  NUMERIC(12,2));
